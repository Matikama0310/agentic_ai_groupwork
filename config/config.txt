"""
Build and deployment automation.
"""

.PHONY: help test test-unit test-integration test-coverage deploy logs clean lint format

help:
	@echo "Agentic Underwriting System - Available Commands"
	@echo "=================================================="
	@echo ""
	@echo "Testing:"
	@echo "  make test                 - Run all tests"
	@echo "  make test-unit            - Run unit tests only"
	@echo "  make test-integration     - Run integration tests"
	@echo "  make test-coverage        - Run tests with coverage report"
	@echo ""
	@echo "Code Quality:"
	@echo "  make lint                 - Lint code with pylint"
	@echo "  make format               - Format code with black/isort"
	@echo ""
	@echo "Deployment:"
	@echo "  make deploy ENV=dev       - Deploy to DEV environment"
	@echo "  make deploy ENV=prod      - Deploy to PROD environment"
	@echo "  make logs ENV=dev         - Tail logs from AWS"
	@echo ""
	@echo "Utilities:"
	@echo "  make clean                - Remove build artifacts"
	@echo "  make docs                 - Generate documentation"
	@echo ""

install:
	pip install -r requirements.txt

test:
	pytest tests/ -v --tb=short

test-unit:
	pytest tests/unit/ -v

test-integration:
	pytest tests/integration/ -v

test-performance:
	pytest tests/performance/ -v

test-coverage:
	pytest tests/ --cov=src --cov-report=term-missing --cov-report=html --cov-fail-under=80
	@echo "Coverage report generated: htmlcov/index.html"

lint:
	pylint src/ lambda/ --disable=C0114,C0115,C0116

format:
	black src/ tests/ lambda/
	isort src/ tests/ lambda/

clean:
	find . -type d -name __pycache__ -exec rm -rf {} +
	find . -type f -name "*.pyc" -delete
	rm -rf .pytest_cache .coverage htmlcov/ dist/ build/

docs:
	cd docs && sphinx-build -b html . _build/html

deploy:
	@if [ -z "$(ENV)" ]; then echo "Usage: make deploy ENV=dev|prod"; exit 1; fi
	sam build
	sam deploy --stack-name agentic-underwriting-$(ENV) --parameter-overrides Environment=$(ENV)
	@echo "Deployment to $(ENV) complete"

logs:
	@if [ -z "$(ENV)" ]; then echo "Usage: make logs ENV=dev|prod"; exit 1; fi
	aws logs tail /aws/lambda/submission-handler-$(ENV) --follow

local-server:
	uvicorn src.api.main:app --reload --port 8000

"""
AWS SAM (Serverless Application Model) template for deployment.
"""

AWSTemplateFormatVersion: '2010-09-09'
Transform: AWS::Serverless-2016-10-31

Description: 'Agentic Underwriting System - Lambda + API Gateway'

Parameters:
  Environment:
    Type: String
    Default: dev
    AllowedValues:
      - dev
      - staging
      - prod
    Description: Environment name
  
  ANTHROPICAPIKey:
    Type: String
    NoEcho: true
    Description: Anthropic API key
  
  LogLevel:
    Type: String
    Default: INFO
    AllowedValues:
      - DEBUG
      - INFO
      - WARNING
      - ERROR

Globals:
  Function:
    Timeout: 60
    MemorySize: 512
    Runtime: python3.12
    Environment:
      Variables:
        ENVIRONMENT: !Ref Environment
        LOG_LEVEL: !Ref LogLevel
        ANTHROPIC_API_KEY: !Ref ANTHROPICAPIKey
        MOCK_EXTERNAL_APIS: 'true'

Resources:
  # ===== Lambda Functions =====
  
  SubmissionHandlerFunction:
    Type: AWS::Serverless::Function
    Properties:
      FunctionName: !Sub 'submission-handler-${Environment}'
      CodeUri: .
      Handler: lambda.submission_handler.lambda_handler
      Description: Process new underwriting submissions
      Policies:
        - Version: '2012-10-17'
          Statement:
            - Effect: Allow
              Action:
                - logs:CreateLogGroup
                - logs:CreateLogStream
                - logs:PutLogEvents
              Resource: arn:aws:logs:*:*:*
            - Effect: Allow
              Action:
                - s3:GetObject
                - s3:PutObject
              Resource: !Sub 'arn:aws:s3:::quotes-${Environment}-*/*'
      Events:
        ApiEvent:
          Type: Api
          Properties:
            RestApiId: !Ref SubmissionApi
            Path: /submit
            Method: POST
      Tags:
        Environment: !Ref Environment
        Application: agentic-underwriting

  OverrideHandlerFunction:
    Type: AWS::Serverless::Function
    Properties:
      FunctionName: !Sub 'override-handler-${Environment}'
      CodeUri: .
      Handler: lambda.override_handler.lambda_handler
      Description: Handle human overrides to AI decisions
      Policies:
        - Version: '2012-10-17'
          Statement:
            - Effect: Allow
              Action:
                - logs:CreateLogGroup
                - logs:CreateLogStream
                - logs:PutLogEvents
              Resource: arn:aws:logs:*:*:*
      Events:
        ApiEvent:
          Type: Api
          Properties:
            RestApiId: !Ref SubmissionApi
            Path: /override
            Method: POST
      Tags:
        Environment: !Ref Environment
        Application: agentic-underwriting

  QueryHandlerFunction:
    Type: AWS::Serverless::Function
    Properties:
      FunctionName: !Sub 'query-handler-${Environment}'
      CodeUri: .
      Handler: lambda.query_handler.lambda_handler
      Description: Query submission status
      Policies:
        - Version: '2012-10-17'
          Statement:
            - Effect: Allow
              Action:
                - logs:CreateLogGroup
                - logs:CreateLogStream
                - logs:PutLogEvents
              Resource: arn:aws:logs:*:*:*
      Events:
        ApiEvent:
          Type: Api
          Properties:
            RestApiId: !Ref SubmissionApi
            Path: /status
            Method: GET
      Tags:
        Environment: !Ref Environment
        Application: agentic-underwriting

  # ===== API Gateway =====
  
  SubmissionApi:
    Type: AWS::Serverless::Api
    Properties:
      Name: !Sub 'agentic-underwriting-api-${Environment}'
      StageName: !Ref Environment
      TracingEnabled: true
      MethodSettings:
        - ResourcePath: '/*'
          HttpMethod: '*'
          LoggingLevel: INFO
          DataTraceEnabled: false
          MetricsEnabled: true

  # ===== CloudWatch Logs =====

    Type: AWS::Logs::LogGroup

      LogGroupName: !Sub '/aws/lambda/override-handler-${Environment}'
      RetentionInDays: 30
