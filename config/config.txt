# UNDERWRITING GUIDELINES (Hard Rules)

# These rules are enforced in src/tools/decision_logic.py

Rule R001: Minimum Credit Score >= 500
Rule R002: Maximum Loss Ratio < 80%

Required Documents:
  - application_form
  - financial_statements
  - loss_history

# ---------------------------------------------------------------------------
# WORKFLOW PHASES
# ---------------------------------------------------------------------------
# Phase 1: Ingestion & Triage
#   START -> ingest_and_classify -> check_data_completeness
#     -> [is_data_complete?]
#         missing_docs -> draft_missing_info -> END
#         data_complete -> Phase 2
#
# Phase 2: Qualification
#   -> enrichment -> check_knockout_rules
#     -> [knockout_check?]
#         fail -> draft_decline -> END
#         pass -> risk_assessment -> Phase 3
#
# Phase 3: The Workbench
#   -> human_checkpoint
#     -> [human_decision?]
#         approve -> generate_quote -> END
#         decline -> draft_decline -> END
#         modify  -> update_state -> risk_assessment (loop back)

# ---------------------------------------------------------------------------
# DECISION OUTCOMES
# ---------------------------------------------------------------------------
# QUOTED       - All rules pass, risk assessed, quote generated
# DECLINED     - Hard rule failure, decline letter with citations
# MISSING_INFO - Critical documents absent, request email sent
# MANUAL_REVIEW - Human modifies, loops back for re-assessment
